# Maximum Subarray

/**
 * Maximum Subarray
 *
 * Problem: https://leetcode.com/problems/maximum-subarray/
 *
 * Given an integer array nums, find the subarray with the largest sum, and return its sum.
 *
 * Example 1:
 * Input: nums = [-2,1,-3,4,-1,2,1,-5,4]
 * Output: 6
 * Explanation: The subarray [4,-1,2,1] has the largest sum 6.
 *
 * Example 2:
 * Input: nums = [1]
 * Output: 1
 *
 * Example 3:
 * Input: nums = [5,4,-1,7,8]
 * Output: 23
 *
 * Constraints:
 * - 1 <= nums.length <= 10^5
 * - -10^4 <= nums[i] <= 10^4
 *
 * Solution Approach:
 * 1. Kadane's Algorithm - One pass solution
 * 2. Keep track of current sum and maximum sum
 * 3. Reset current sum to 0 if it becomes negative
 * 4. Update maximum sum at each step
 *
 * Time Complexity: O(n) where n is the length of nums array
 * Space Complexity: O(1) as we only use a constant amount of extra space
 */

/**
 * Maximum Subarray - Kadane's Algorithm
 *
 * Thu·∫≠t to√°n Kadane - Gi·∫£i ph√°p m·ªôt l·∫ßn duy·ªát
 *
 * @param nums - M·∫£ng s·ªë nguy√™n
 * @returns T·ªïng l·ªõn nh·∫•t c·ªßa subarray
 */
function maxSubArray(nums: number[]): number {
  if (nums.length === 0) return 0;

  let currentSum = nums[0]; // T·ªïng hi·ªán t·∫°i
  let maxSum = nums[0]; // T·ªïng l·ªõn nh·∫•t

  for (let i = 1; i < nums.length; i++) {
    // C·∫≠p nh·∫≠t t·ªïng hi·ªán t·∫°i: l·∫•y max gi·ªØa s·ªë hi·ªán t·∫°i v√† t·ªïng hi·ªán t·∫°i + s·ªë hi·ªán t·∫°i
    currentSum = Math.max(nums[i], currentSum + nums[i]);

    // C·∫≠p nh·∫≠t t·ªïng l·ªõn nh·∫•t
    maxSum = Math.max(maxSum, currentSum);
  }

  return maxSum;
}

/**
 * Alternative Solution: Divide and Conquer Approach
 *
 * Gi·∫£i ph√°p thay th·∫ø: Chia ƒë·ªÉ tr·ªã
 *
 * @param nums - M·∫£ng s·ªë nguy√™n
 * @returns T·ªïng l·ªõn nh·∫•t c·ªßa subarray
 */
function maxSubArrayDivideAndConquer(nums: number[]): number {
  if (nums.length === 0) return 0;
  if (nums.length === 1) return nums[0];

  return divideAndConquer(nums, 0, nums.length - 1);
}

function divideAndConquer(nums: number[], left: number, right: number): number {
  if (left === right) return nums[left];

  const mid = Math.floor((left + right) / 2);

  // T√¨m t·ªïng l·ªõn nh·∫•t ·ªü n·ª≠a tr√°i
  const leftMax = divideAndConquer(nums, left, mid);

  // T√¨m t·ªïng l·ªõn nh·∫•t ·ªü n·ª≠a ph·∫£i
  const rightMax = divideAndConquer(nums, mid + 1, right);

  // T√¨m t·ªïng l·ªõn nh·∫•t ƒëi qua gi·ªØa
  const crossMax = maxCrossingSubarray(nums, left, mid, right);

  return Math.max(leftMax, rightMax, crossMax);
}

function maxCrossingSubarray(
  nums: number[],
  left: number,
  mid: number,
  right: number
): number {
  // T√¨m t·ªïng l·ªõn nh·∫•t t·ª´ mid v·ªÅ tr√°i
  let leftSum = -Infinity;
  let sum = 0;
  for (let i = mid; i >= left; i--) {
    sum += nums[i];
    leftSum = Math.max(leftSum, sum);
  }

  // T√¨m t·ªïng l·ªõn nh·∫•t t·ª´ mid + 1 v·ªÅ ph·∫£i
  let rightSum = -Infinity;
  sum = 0;
  for (let i = mid + 1; i <= right; i++) {
    sum += nums[i];
    rightSum = Math.max(rightSum, sum);
  }

  return leftSum + rightSum;
}

/**
 * Brute Force Solution (for comparison)
 *
 * Gi·∫£i ph√°p v√©t c·∫°n (ƒë·ªÉ so s√°nh)
 *
 * @param nums - M·∫£ng s·ªë nguy√™n
 * @returns T·ªïng l·ªõn nh·∫•t c·ªßa subarray
 */
function maxSubArrayBruteForce(nums: number[]): number {
  let maxSum = -Infinity;

  for (let i = 0; i < nums.length; i++) {
    let currentSum = 0;
    for (let j = i; j < nums.length; j++) {
      currentSum += nums[j];
      maxSum = Math.max(maxSum, currentSum);
    }
  }

  return maxSum;
}

/**
 * Solution with Subarray Indices (for debugging)
 *
 * Gi·∫£i ph√°p tr·∫£ v·ªÅ c·∫£ ch·ªâ s·ªë c·ªßa subarray (ƒë·ªÉ debug)
 *
 * @param nums - M·∫£ng s·ªë nguy√™n
 * @returns Object ch·ª©a t·ªïng l·ªõn nh·∫•t v√† ch·ªâ s·ªë c·ªßa subarray
 */
function maxSubArrayWithIndices(nums: number[]): {
  maxSum: number;
  start: number;
  end: number;
} {
  if (nums.length === 0) return { maxSum: 0, start: -1, end: -1 };

  let currentSum = nums[0];
  let maxSum = nums[0];
  let start = 0;
  let end = 0;
  let tempStart = 0;

  for (let i = 1; i < nums.length; i++) {
    if (nums[i] > currentSum + nums[i]) {
      // B·∫Øt ƒë·∫ßu subarray m·ªõi
      currentSum = nums[i];
      tempStart = i;
    } else {
      // Ti·∫øp t·ª•c subarray hi·ªán t·∫°i
      currentSum += nums[i];
    }

    if (currentSum > maxSum) {
      maxSum = currentSum;
      start = tempStart;
      end = i;
    }
  }

  return { maxSum, start, end };
}
