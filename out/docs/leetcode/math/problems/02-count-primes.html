  • Count Primes
  • Problem: https://leetcode.com/problems/count-primes/
  • Given an integer n, return the number of prime numbers that are strictly less than n.
  • Example 1:
  • Input: n = 10
  • Output: 4
  • Explanation: There are 4 prime numbers less than 10, they are 2, 3, 5, 7.
  • Example 2:
  • Input: n = 0
  • Output: 0
  • Example 3:
  • Input: n = 1
  • Output: 0

    • 0 <= n <= 5 * 10^6
  • Solution Approach:
    1. Use Sieve of Eratosthenes algorithm
    1. Create a boolean array to mark non-prime numbers
    1. Start from 2 and mark all multiples as non-prime
    1. Count the remaining unmarked numbers

/**

/**

  • Gi·∫£i ph√°p v√©t c·∫°n (ƒë·ªÉ so s√°nh)

/**

  • Ki·ªÉm tra xem m·ªôt s·ªë c√≥ ph·∫£i l√† s·ªë nguy√™n t·ªë kh√¥ng

/**

  • Solution with Prime List Generation
  • Gi·∫£i ph√°p v·ªõi t·∫°o danh s√°ch s·ªë nguy√™n t·ªë
  • @param n - S·ªë nguy√™n d∆∞∆°ng

/**

  • Segmented Sieve (for very large numbers)
  • S√†ng ph√¢n ƒëo·∫°n (cho s·ªë r·∫•t l·ªõn)
  • @param n - S·ªë nguy√™n d∆∞∆°ng
  • @returns S·ªë l∆∞·ª£ng s·ªë nguy√™n t·ªë nh·ªè h∆°n n */ function countPrimesSegmented(n: number): number { if (n <= 2) return 0;

const segmentSize = Math.floor(Math.sqrt(n)) + 1; const isPrime = new Array(segmentSize).fill(true); isPrime[0] = false; isPrime[1] = false;

// T√¨m c√°c s·ªë nguy√™n t·ªë nh·ªè h∆°n cƒÉn b·∫≠c hai c·ªßa n const smallPrimes: number[] = []; for (let i = 2; i < segmentSize; i++) { if (isPrime[i]) { smallPrimes.push(i); for (let j = i * i; j < segmentSize; j += i) { isPrime[j] = false; } } }

let count = smallPrimes.length;

// X·ª≠ l√Ω t·ª´ng ƒëo·∫°n for (let low = segmentSize; low < n; low += segmentSize) { const high = Math.min(low + segmentSize, n); const segment = new Array(high - low).fill(true);

for (const prime of smallPrimes) {
  const start = Math.max(prime * prime, Math.ceil(low / prime) * prime);
  for (let j = start; j < high; j += prime) {
    segment[j - low] = false;
  }
}

for (let i = 0; i < segment.length; i++) {
  if (segment[i]) {
    count++;
  }
}

}

return count; }
