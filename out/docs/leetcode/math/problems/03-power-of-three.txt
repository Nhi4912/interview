/**
 * Power of Three
 *
 * Problem: https://leetcode.com/problems/power-of-three/
 *
 * Given an integer n, return true if it is a power of three. Otherwise, return false.
 * An integer n is a power of three, if there exists an integer x such that n == 3^x.
 *
 * Example 1:
 * Input: n = 27
 * Output: true
 * Explanation: 27 = 3^3
 *
 * Example 2:
 * Input: n = 0
 * Output: false
 * Explanation: There is no x where 3^x = 0.
 *
 * Example 3:
 * Input: n = -1
 * Output: false
 * Explanation: There is no x where 3^x = (-1).
 *
 * Constraints:
 * - -2^31 <= n <= 2^31 - 1
 *
 * Solution Approach:
 * 1. Iterative division: Keep dividing by 3 until we can't
 * 2. Mathematical approach: Use logarithm
 * 3. Bit manipulation: Check if it's the only set bit in base-3
 * 4. Precomputed approach: Use known powers of 3
 *
 * Time Complexity: O(log n) for iterative, O(1) for mathematical
 * Space Complexity: O(1) for all approaches
 */

/**
 * Power of Three - Iterative Division
 *
 * Gi·∫£i ph√°p chia l·∫∑p cho b√†i to√°n L≈©y th·ª´a c·ªßa 3
 *
 * @param n - S·ªë nguy√™n c·∫ßn ki·ªÉm tra
 * @returns true n·∫øu n l√† l≈©y th·ª´a c·ªßa 3, false n·∫øu kh√¥ng
 */
function isPowerOfThree(n: number): boolean {
  if (n <= 0) return false;

  // Chia li√™n ti·∫øp cho 3 cho ƒë·∫øn khi kh√¥ng th·ªÉ chia ƒë∆∞·ª£c n·ªØa
  while (n % 3 === 0) {
    n = Math.floor(n / 3);
  }

  // N·∫øu k·∫øt qu·∫£ cu·ªëi c√πng l√† 1, th√¨ n l√† l≈©y th·ª´a c·ªßa 3
  return n === 1;
}

/**
 * Alternative Solution: Mathematical Approach
 *
 * Gi·∫£i ph√°p thay th·∫ø: Ph∆∞∆°ng ph√°p to√°n h·ªçc
 *
 * @param n - S·ªë nguy√™n c·∫ßn ki·ªÉm tra
 * @returns true n·∫øu n l√† l≈©y th·ª´a c·ªßa 3, false n·∫øu kh√¥ng
 */
function isPowerOfThreeMath(n: number): boolean {
  if (n <= 0) return false;

  // S·ª≠ d·ª•ng logarit: n = 3^x => x = log3(n) = log(n) / log(3)
  const log3 = Math.log(n) / Math.log(3);

  // Ki·ªÉm tra xem log3 c√≥ ph·∫£i l√† s·ªë nguy√™n kh√¥ng
  return Math.abs(log3 - Math.round(log3)) < 1e-10;
}

/**
 * Solution with Precomputed Powers
 *
 * Gi·∫£i ph√°p v·ªõi l≈©y th·ª´a ƒë∆∞·ª£c t√≠nh tr∆∞·ªõc
 *
 * @param n - S·ªë nguy√™n c·∫ßn ki·ªÉm tra
 * @returns true n·∫øu n l√† l≈©y th·ª´a c·ªßa 3, false n·∫øu kh√¥ng
 */
function isPowerOfThreePrecomputed(n: number): boolean {
  if (n <= 0) return false;

  // C√°c l≈©y th·ª´a c·ªßa 3 trong ph·∫°m vi 32-bit integer
  const powersOfThree = new Set([
    1, 3, 9, 27, 81, 243, 729, 2187, 6561, 19683, 59049, 177147, 531441,
    1594323, 4782969, 14348907, 43046721, 129140163, 387420489, 1162261467,
  ]);

  return powersOfThree.has(n);
}

/**
 * Recursive Solution
 *
 * Gi·∫£i ph√°p ƒë·ªá quy
 *
 * @param n - S·ªë nguy√™n c·∫ßn ki·ªÉm tra
 * @returns true n·∫øu n l√† l≈©y th·ª´a c·ªßa 3, false n·∫øu kh√¥ng
 */
function isPowerOfThreeRecursive(n: number): boolean {
  if (n <= 0) return false;
  if (n === 1) return true;
  if (n % 3 !== 0) return false;

  return isPowerOfThreeRecursive(Math.floor(n / 3));
}

/**
 * Solution with Power Generation
 *
 * Gi·∫£i ph√°p v·ªõi t·∫°o l≈©y th·ª´a
 *
 * @param n - S·ªë nguy√™n c·∫ßn ki·ªÉm tra
 * @returns Object ch·ª©a k·∫øt qu·∫£ v√† l≈©y th·ª´a t∆∞∆°ng ·ª©ng
 */
function isPowerOfThreeWithExponent(n: number): {
  isPower: boolean;
  exponent: number | null;
} {
  if (n <= 0) return { isPower: false, exponent: null };
  if (n === 1) return { isPower: true, exponent: 0 };

  let exponent = 0;
  let current = n;

  while (current % 3 === 0) {
    current = Math.floor(current / 3);
    exponent++;
  }

  if (current === 1) {
    return { isPower: true, exponent: exponent };
  } else {
    return { isPower: false, exponent: null };
  }
}

/**
 * Solution with All Powers Generation
 *
 * Gi·∫£i ph√°p v·ªõi t·∫°o t·∫•t c·∫£ l≈©y th·ª´a
 *
 * @param maxPower - L≈©y th·ª´a t·ªëi ƒëa c·∫ßn t·∫°o
 * @returns M·∫£ng c√°c l≈©y th·ª´a c·ªßa 3
 */
function generatePowersOfThree(maxPower: number): number[] {
  const powers: number[] = [];
  let power = 1;

  for (let i = 0; i <= maxPower; i++) {
    powers.push(power);
    power *= 3;

    // Ki·ªÉm tra overflow
    if (power > Number.MAX_SAFE_INTEGER) {
      break;
    }
  }

  return powers;
}

/**
 * Solution with Range Check
 *
 * Gi·∫£i ph√°p v·ªõi ki·ªÉm tra ph·∫°m vi
 *
 * @param n - S·ªë nguy√™n c·∫ßn ki·ªÉm tra
 * @returns true n·∫øu n l√† l≈©y th·ª´a c·ªßa 3, false n·∫øu kh√¥ng
 */
function isPowerOfThreeRange(n: number): boolean {
  if (n <= 0) return false;

  // L≈©y th·ª´a l·ªõn nh·∫•t c·ªßa 3 trong ph·∫°m vi 32-bit integer
  const maxPowerOfThree = 1162261467; // 3^19

  // N·∫øu n kh√¥ng chia h·∫øt cho maxPowerOfThree, th√¨ n kh√¥ng ph·∫£i l≈©y th·ª´a c·ªßa 3
  return maxPowerOfThree % n === 0;
}
